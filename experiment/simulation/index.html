Aim : Write a program that takes two operands and one operator from the user and perform the operation and prints the result by using switch statement.

Simulation:

=

Instruction:

  1. Enter the 2 number in given column.
  2. Select the operator from the given option in operator section.
  3. Click on the Calculate button.
  4. After calculation value will be displayed in the Value section.

Code:

#include<stdio.h>
int main() {
char operator;
double first, second;
int m;
printf("Enter an operator (+, -, *,/,mod): ");
scanf("%c",&operator);
printf("Enter two operands: ");
scanf("%lf %lf", &first, &second);
switch (operator)
{
  case '+':
    printf("%.3lf",first + second);
    break;
  case '-':
    printf("%.3lf",first - second);
    break;
  case '*':
    printf("%.3lf",first * second);
    break;
  case '/':
    printf("%.3lf",first / second);
    break;
  case '%':
    m=(int)(first)%(int)(second);
    printf("%d",m);
    break;
  default:
   printf("Error! operator is not correct");
}
return 0;
}
